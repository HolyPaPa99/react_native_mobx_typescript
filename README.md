| |--log                 #日志格式化组件目录
| |--mobx                #mobx配置目录
| |--navigation          #导航配置目录
| |--utils               #公共工具组件目录
|--components            #UI组件目录
| |--form                #表单组件目录
| |--languageSwitch      #语言切换组件目录
| |--navigator           #导航组件目录
|--config                #系统配置目录
|--models                #数据模型定义目录
| |--user                #用户数据模型
|--screens               #页面开发目录
|--home                  #主页
